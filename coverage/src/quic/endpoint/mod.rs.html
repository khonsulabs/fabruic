508#[async_trait]
509pub trait Dangerous {
510 /// Establishes a new [`Connection`](crate::Connection) to a server without
511 /// verifying the servers [`Certificate`]. The servers
512 /// [`CertificateChain`](crate::CertificateChain) can still be manually
513 /// insepcted through
514 /// [`Connection::peer_identity`](crate::Connection::peer_identity).
515 ///
516 /// See [`connect`](Endpoint::connect) for more information on host name
517 /// resolution.
518 ///
519 /// # Notes
520 /// A client certificate [`KeyPair`] set with
521 /// [`Builder::set_client_key_pair`] will be ignored, use `client_key_pair`
522 /// to add a client certificate to this connection.
523 ///
524 /// # Safety
525 /// Connecting to a server without verifying the [`Certificate`] provides no
526 /// way for the client to authenticate the servers identity.
527 /// This is primarily used to enable connections to unknown user-hosted
528 /// servers, e.g. multiplayer.
529 ///
530 /// There are many ways to prevent the need for this feature in certain
531 /// situations:
532 /// - during testing, a temporary certificate can be created
533 /// - use [Let's Encrypt](https://en.wikipedia.org/wiki/Let%27s_Encrypt) to
534 /// get a free certificate if a domain is present
535 /// - provide a middle-man service that helps connect clients with servers
536 /// by automatically communicating the servers public key
537 /// - share a public key over third-party communication channels beforehand
538 /// as a last resort
539 ///
540 /// # Errors
541 /// - [`error::Connect::ParseUrl`] if the URL couldn't be parsed
542 /// - [`error::Connect::Port`] if the URL didn't contain a port
543 /// - [`error::Connect::ParseDomain`] if the domain couldn't be parsed
544 /// - [`error::Connect::TrustDns`] if the URL couldn't be resolved to an IP
545 /// address with [`trust-dns`](trust_dns_resolver)
546 /// - [`error::Connect::StdDns`] if the URL couldn't be resolved to an IP
547 /// address with [`ToSocketAddrs`]
548 /// - [`error::Connect::NoIp`] if no IP address was found for that domain
549 ///
550 /// # Examples
551 /// ```
552 /// # #[tokio::main] async fn main() -> anyhow::Result<()> {
553 /// use fabruic::{dangerous, Endpoint};
554 ///
555 /// let endpoint = Endpoint::new_client()?;
556 /// let connecting =
557 /// dangerous::Endpoint::connect_unverified(&endpoint, "quic://localhost:443", None).await?;
558 /// # Ok(()) }
559 /// ```
560 async fn connect_unverified<U: AsRef<str> + Send>(
561 endpoint: &Self,
562 url: U,
563 client_key_pair: Option<KeyPair>,
564 ) -> Result<Connecting, error::Connect>;
